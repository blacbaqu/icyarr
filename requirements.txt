# ============================================================
# ICYARR BACKEND — requirements.txt
# ============================================================
# These are the Python dependencies required for the FastAPI
# backend, M3U loading, stream testing, and SQLite persistence.
# ============================================================

# FastAPI framework for building the backend API
fastapi

# Uvicorn ASGI server used to run FastAPI
uvicorn

# HTTP client used for fetching M3U playlists and testing streams
requests

# Data validation and request models (M3URequest, StreamRequest, etc.)
pydantic

# SQLite binary package — ensures SQLite works inside Python slim images
# (Python includes sqlite3, but this prevents missing extension issues)
sqlite3-binary
